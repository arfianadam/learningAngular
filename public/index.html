Create list

HTML


<div ng-app="todoApp">
    <div ng-controller="ListCtrl">
        <form class="form-group" ng-submit="addList()">
            <label for="">Add a todo</label>
            <input type="text" ng-model="enteredList" class="form-control">
        </form>
        <h3>My todo list</h3>
        <ul>
            <li ng-repeat="todo in todos">{ { todo } }</li>
        </ul>
    </div>
</div>

JS


angular.module('todoApp', [])
    .controller('ListCtrl', ['$scope', function($scope) {
            $scope.todos = [];
            $scope.addList = function() {
                $scope.todos.push($scope.enteredList);
                $scope.enteredList = '';
            };
        }]);
